Example

// Example > With included views
var data = {
  firstName: "John",
  lastName:  "Doe"
};

// declare views
var demo = new Gumok.view("#example", data, {
  index: function() {
    document.write("<h2>Executing the view</h2>");
  },
  other: function(data) {
    document.write(data);
  }
});
